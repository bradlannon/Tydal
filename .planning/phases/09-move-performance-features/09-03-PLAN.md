Also add automation to the drum track object: `automation: Array.from({ length: 16 }, () => null)`.

Export a helper:
```javascript
export function setStepAutomation(trackId, step, paramName, value) {
  const track = tracks[trackId];
  if (!track || step < 0 || step >= 16) return;
  track.automation[step] = { paramName, value };
  document.dispatchEvent(new CustomEvent('automation-update', { detail: { trackId, step } }));
}

export function getStepAutomation(trackId, step) {
  const track = tracks[trackId];
  if (!track || step < 0 || step >= 16) return null;
  return track.automation[step];
}

export function clearStepAutomation(trackId, step) {
  const track = tracks[trackId];
  if (!track || step < 0 || step >= 16) return;
  track.automation[step] = null;
  document.dispatchEvent(new CustomEvent('automation-update', { detail: { trackId, step } }));
}
```

**Automation playback in melodic-sequencer.js:**

In the Tone.Sequence callback, after the existing note trigger loop, apply automation for each track at the current step:
```javascript
for (const track of tracks) {
  if (track.type !== 'melodic' || track.muted) continue;

  // Apply per-step automation if present
  const auto = track.automation[step];
  if (auto) {
    _applyAutomation(track, auto, time);
  }

  for (const note of track.grid[step]) {
    track.synth.triggerAttackRelease(note, '16n', time, 0.7);
  }
}
```

The `_applyAutomation` helper maps param names to synth/effects actions:
```javascript
function _applyAutomation(track, auto, time) {
  const { paramName, value } = auto;
  switch (paramName) {
    case 'Cutoff':
      track.synth.set({ filter: { frequency: value } });
      break;
    case 'Res':
      track.synth.set({ filter: { Q: value } });
      break;
    case 'Reverb':
      track.effectsChain.reverb.wet.value = value;
      break;
    case 'Delay':
      track.effectsChain.delay.wet.value = value;
      break;
    case 'Attack':
      track.synth.set({ envelope: { attack: value } });
      break;
    case 'Release':
      track.synth.set({ envelope: { release: value } });
      break;
    // Volume and other params can be added as needed
  }
}
```

Apply automation BEFORE triggering the notes so the synth plays with the automated value. This is important — set params then trigger.

**Step button refactor for hold vs tap:**

Replace the current simple pointerdown → toggleStep with:
```javascript
function _onStepPointerDown(e) {
  e.preventDefault();
  const step = parseInt(e.currentTarget.dataset.step);
  _heldStep = step;

  _holdTimer = setTimeout(() => {
    // If we reach here, it's a hold — don't toggle, just hold for automation
    _holdTimer = null;
  }, 200);

  // Add release listeners
  e.currentTarget.addEventListener('pointerup', _onStepPointerUp, { once: true });
  e.currentTarget.addEventListener('pointerleave', _onStepPointerUp, { once: true });
}

function _onStepPointerUp(e) {
  const step = parseInt(e.currentTarget.dataset.step);
  if (_holdTimer !== null) {
    // Released before 200ms — this is a tap, do the normal toggle
    clearTimeout(_holdTimer);
    _holdTimer = null;
    _doStepTap(step);
  }
  _heldStep = -1;
}
```

Move the existing tap logic into `_doStepTap(step)`.
